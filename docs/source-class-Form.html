 50:     public function render()
 51:     {
 52:         $prop = array();
 53:         $event = array();
 54:         foreach ($this->attributes as $k => $v) {
 55:             if (is_int($k)) {
 56:                 if (!empty($v)) {
 57:                     $prop[$v] = $v;
 58:                 }
 59:             } elseif (in_array($k, array('checked', 'readonly'))) {
 60:                 if ($v) {
 61:                     $prop[$k] = $k;
 62:                 }
 63:             } elseif (in_array($k, array('result'))) {
 64:                 $prop[$k] = 'data-'.$k.'="'.$v.'"';
 65:             } elseif (preg_match('/^on([a-z]+)/', $k, $match)) {
 66:                 $event[$match[1]] = $v;
 67:             } else {
 68:                 if (!in_array($k, array('itemClass', 'labelClass', 'label', 'comment', 'value', 'dataPreview', 'previewSrc', 'options', 'optgroup', 'validator'))) {
 69:                     if (is_int($v) || is_bool($v)) {
 70:                         $prop[$k] = $k.'='.$v;
 71:                     } else {
 72:                         $prop[$k] = $k.'="'.$v.'"';
 73:                     }
 74:                 }
 75:                 $$k = $v;
 76:             }
 77:         }
 78:         if (isset($id) && empty($name)) {
 79:             $name = $id;
 80:             $prop['name'] = 'name="'.$name.'"';
 81:         }
 82:         if (isset($id)) {
 83:             if (isset($validator)) {
 84:                 $js = array();
 85:                 $js[] = '"'.$id.'"';
 86:                 $js[] = '"'.$validator[0].'"';
 87:                 $js[] = $validator[1];
 88:                 if (isset($validator[2])) {
 89:                     $js[] = '"'.$validator[2].'"';
 90:                     $js[] = empty($validator[3]) || $validator[3] === null ? 'null' : '"'.$validator[3].'"';
 91:                     $js[] = '"'.\Html::$form->attributes['id'].'"';
 92:                 }
 93:                 $this->javascript[] = 'new GValidator('.implode(', ', $js).');';
 94:                 unset($validator);
 95:             }
 96:             foreach ($event as $on => $func) {
 97:                 $this->javascript[] = '$G("'.$id.'").addEvent("'.$on.'", '.$func.');';
 98:             }
 99:         }
100:         if ($this->tag == 'select') {
101:             unset($prop['type']);
102:             $value = isset($value) ? $value : null;
103:             if (isset($options)) {
104:                 $datas = array();
105:                 foreach ($options as $k => $v) {
106:                     $sel = $value == $k ? ' selected' : '';
107:                     $datas[] = '<option value="'.$k.'"'.$sel.'>'.$v.'</option>';
108:                 }
109:                 $value = implode('', $datas);
110:             } elseif (isset($optgroup)) {
111:                 $datas = array();
112:                 foreach ($optgroup as $group_label => $options) {
113:                     $datas[] = '<optgroup label="'.$group_label.'">';
114:                     foreach ($options as $k => $v) {
115:                         $sel = $value == $k ? ' selected' : '';
116:                         $datas[] = '<option value="'.$k.'"'.$sel.'>'.$v.'</option>';
117:                     }
118:                     $datas[] = '</optgroup>';
119:                 }
120:                 $value = implode('', $datas);
121:             }
122:         } elseif (isset($value)) {
123:             if ($this->tag == 'textarea') {
124:                 $value = str_replace(array('{', '}', '&amp;'), array('&#x007B;', '&#x007D;', '&'), htmlspecialchars($value));
125:             } else {
126:                 if (is_int($value) || is_bool($value)) {
127:                     $prop['value'] = 'value='.$value;
128:                 } else {
129:                     $prop['value'] = 'value="'.$value.'"';
130:                 }
131:             }
132:         }
133:         if (!empty($comment) && empty($prop['title'])) {
134:             $prop['title'] = 'title="'.$comment.'"';
135:         }
136:         if (isset($dataPreview)) {
137:             $prop['data-preview'] = 'data-preview="'.$dataPreview.'"';
138:         }
139:         $prop = implode(' ', $prop);
140:         if ($this->tag == 'input') {
141:             $element = '<'.$this->tag.' '.$prop.'>';
142:         } elseif (isset($value)) {
143:             $element = '<'.$this->tag.' '.$prop.'>'.$value.'</'.$this->tag.'>';
144:         } else {
145:             $element = '<'.$this->tag.' '.$prop.'></'.$this->tag.'>';
146:         }
147:         if (empty($itemClass)) {
148:             $input = empty($comment) ? '' : '<div class="item">';
149:             if (empty($labelClass) && empty($label)) {
150:                 $input .= $element;
151:             } elseif (isset($type) && $type == 'checkbox') {
152:                 $input .= '<label'.(empty($labelClass) ? '' : ' class="'.$labelClass.'"').'>'.$element.'&nbsp;'.$label.'</label>';
153:             } else {
154:                 $input .= '<label'.(empty($labelClass) ? '' : ' class="'.$labelClass.'"').'>';
155:                 if (!empty($label)) {
156:                     $input .= $label.(isset($type) && in_array($type, array('checkbox', 'radio')) ? '&nbsp;' : '&nbsp;:&nbsp;');
157:                 }
158:                 $input .=$element.'</label>';
159:             }
160:             if (!empty($comment)) {
161:                 $input .= '<div class="comment"'.(empty($id) ? '' : ' id="result_'.$id.'"').'>'.$comment.'</div></div>';
162:             }
163:         } else {
164:             $input = '<div class="'.$itemClass.'">';
165:             if (isset($type) && $type == 'checkbox') {
166:                 $input .= '<label'.(empty($labelClass) ? '' : ' class="'.$labelClass.'"').'>'.$element.'&nbsp;'.$label.'</label>';
167:             } else {
168:                 if (isset($dataPreview)) {
169:                     $input .= '<div class=usericon><span><img src="'.$previewSrc.'" alt="Image preview" id='.$dataPreview.'></span></div>';
170:                 }
171:                 if (isset($label)) {
172:                     $input .= '<label for="'.$id.'">'.$label.'</label>';
173:                 }
174:                 $input .= '<span'.(empty($labelClass) ? '' : ' class="'.$labelClass.'"').'>'.$element.'</span>';
175:             }
176:             if (!empty($comment)) {
177:                 $input .= '<div class="comment"'.(empty($id) ? '' : ' id="result_'.$id.'"').'>'.$comment.'</div>';
178:             }
179:             $input .= '</div>';
180:         }
181:         return $input;

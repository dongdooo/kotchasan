 54:     {
 55:         $qs = array();
 56:         foreach ($_GET AS $key => $value) {
 57:             if (preg_match('/^_{1,}(.*)$/', $key, $match)) {
 58:                 $key = $match[1];
 59:             }
 60:             $qs[$key] = $key.'='.rawurlencode($value);
 61:         }
 62:         $f = is_array($f) ? $f[2] : $f;
 63:         if (!empty($f)) {
 64:             foreach (explode('&', str_replace('&amp;', '&', $f)) AS $item) {
 65:                 if (preg_match('/^(.*)=(.*)$/', $item, $match)) {
 66:                     $qs[$match[1]] = isset($match[2]) ? $match[1].'='.$match[2] : $match[1];

 68:             }
 69:         }
 70:         return preg_replace('/&amp;id=0/', '', (sizeof($qs) > 0 ? '?'.implode('&amp;', $qs) : ''));

123:         $splitpage = ($start > 2) ? str_replace(':page', 1, $url) : '';
124:         for ($i = $start; $i <= $totalpage && $maxlink > 0; $i++) {
125:             $splitpage .= ($i == $page) ? '<strong>'.$i.'</strong>' : str_replace(':page', $i, $url);
126:             $maxlink--;
127:         }
128:         $splitpage .= ($i < $totalpage) ? str_replace(':page', $totalpage, $url) : '';

172:             $replace[] = '';
173:         } else {
174:             $patt[] = '/{document}/';
175:             $replace[] = $encode ? rawurlencode($document) : $document;
176:         }
177:         $patt[] = '/{module}/';
178:         $replace[] = $encode ? rawurlencode($module) : $module;
179:         if (empty($catid)) {
180:             $patt[] = '/((cat={catid}&amp;)|([\/-]{catid}))/';
181:             $replace[] = '';
182:         } else {
183:             $patt[] = '/{catid}/';
184:             $replace[] = (int)$catid;
185:         }
186:         if (empty($id)) {
187:             $patt[] = '/(((&amp;|\?)id={id})|([\/-]{id}))/';
188:             $replace[] = '';
189:         } else {
190:             $patt[] = '/{id}/';
191:             $replace[] = (int)$id;
192:         }
193:         $link = preg_replace($patt, $replace, \Kotchasan::$urls[\Kotchasan::$config->module_url]);
194:         if (!empty($query)) {
195:             $link = preg_match('/[\?]/u', $link) ? $link.'&amp;'.$query : $link.'?'.$query;
196:         }
197:         return WEB_URL.$link;

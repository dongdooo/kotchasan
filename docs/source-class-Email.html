 75:     public static function custom($mailto, $replyto, $subject, $msg)
 76:     {
 77:         $charset = empty(\Kotchasan::$config->email_charset) ? 'utf-8' : strtolower(\Kotchasan::$config->email_charset);
 78:         if (empty($replyto)) {
 79:             $replyto = array(\Kotchasan::$config->noreply_email, strip_tags(\Kotchasan::$config->web_title));
 80:         } elseif (preg_match('/^(.*)<(.*?)>$/', $replyto, $match)) {
 81:             $replyto = array($match[1], (empty($match[2]) ? $match[1] : $match[2]));
 82:         } else {
 83:             $replyto = array($replyto, $replyto);
 84:         }
 85:         if ($charset !== 'utf-8') {
 86:             $subject = iconv('utf-8', $charset, $subject);
 87:             $msg = iconv('utf-8', $charset, $msg);
 88:             $replyto[1] = iconv('utf-8', $charset, $replyto[1]);
 89:         }
 90:         $messages = array();
 91:         if (empty(\Kotchasan::$config->email_use_phpMailer)) {
 92:             // ส่งอีเมล์ด้วยฟังก์ชั่นของ PHP
 93:             foreach (explode(',', $mailto) as $email) {
 94:                 $headers = "MIME-Version: 1.0\r\n";
 95:                 $headers .= "Content-type: text/html; charset=$charset\r\n";
 96:                 $headers .= "Content-Transfer-Encoding: quoted-printable\r\n";
 97:                 $headers .= "To: $email\r\n";
 98:                 $headers .= "From: $replyto[1]\r\n";
 99:                 $headers .= "Reply-to: $replyto[0]\r\n";
100:                 $headers .= "X-Mailer: PHP mailer\r\n";
101:                 if (!@mail($email, $subject, $msg, $headers)) {
102:                     $messages = array(\Language::get('Unable to send mail'));
103:                 }
104:             }
105:         } else {
106:             // ส่งอีเมล์ด้วย PHPMailer
107:             include_once ROOT_PATH.'core/class.phpmailer.php';
108:             // Create a new PHPMailer instance
109:             $mail = new PHPMailer;
110:             // Tell PHPMailer to use SMTP
111:             $mail->isSMTP();
112:             // charset
113:             $mail->CharSet = $charset;
114:             // use html
115:             $mail->IsHTML();
116:             $mail->SMTPAuth = empty(\Kotchasan::$config->email_SMTPAuth) ? false : true;
117:             if ($mail->SMTPAuth) {

119:                 $mail->Password = \Kotchasan::$config->email_Password;

121:             }

123:                 $mail->Host = \Kotchasan::$config->email_Host;

128:             $mail->AddReplyTo($replyto[0], $replyto[1]);
